    ${CMAKE_CURRENT_SOURCE_DIR}

    ${CMAKE_CURRENT_SOURCE_DIR}/../Core/Inc   # spi.h, usart.h ç­‰
    ${CMAKE_CURRENT_SOURCE_DIR}/examples/porting

cmake_minimum_required(VERSION 3.16)

project(FSM_Core LANGUAGES C CXX)

set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)

set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)


add_library(finite_state_machine STATIC
    HSM_Core.cpp
    HSM_Bus.cpp
)


target_include_directories(finite_state_machine
    PUBLIC
        ${CMAKE_CURRENT_SOURCE_DIR}
    PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/../Log
)

add_subdirectory(static_stack)
add_subdirectory(static_queue)

target_link_libraries(finite_state_machine PRIVATE static_stack static_queue)

add_executable(HSM_Boundary_Test HSM_Boundary_Test.cpp)
target_link_libraries(HSM_Boundary_Test PRIVATE finite_state_machine)

add_executable(HSM_Bus_Demo HSM_Bus_Demo.cpp)
target_link_libraries(HSM_Bus_Demo PRIVATE finite_state_machine)

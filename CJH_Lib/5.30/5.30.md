2. 按键消抖与LED切换
```c
int main()
{
	int keyflag = 0;
	while(1)
	{
		if(key_read_raw(KEY0) != 1 )//按下
		{
			delay_ms(20);
			keyflag = 1;
			if(key_read_raw(KEY0) = 1 )//按下松开
			{
			delay_ms(20);
			keyflag = 0;
			}
		}
		if(keyflag == 0 && key_read_raw(KEY0) = 1 )//取反
		{
			if(gpiopinread(LED0) == 1)
			{
				gpio_pin_write(LED0, 0);
			}
			else
			{
				gpio_pin_write(LED0, 1);
			}
		}
	}
}
```
